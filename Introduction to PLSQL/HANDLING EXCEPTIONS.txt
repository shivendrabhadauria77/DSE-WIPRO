HANDLING EXCEPTIONS

Q1) Create a table MESSAGES table with a column Result of type varchar2(1000)
Write a program that prompt for SAL and get the corresponding ENAME from the EMP table. Do not use Explicit Cursors.
If the value entered return one row then Insert into MESSAGES table with the ENAME ||' '||SAL.
If the value returned is more than one row then Insert into MESSAGES table as 'More than one employee with a salary <>.
If no value is returned then insert into MESSAGES table as
'No Employee with that salary <>.
Any other error insert 'Other Error'.
Use all NAMED EXCEPTION HANDLERS.

SOLUTION) 

1. Create the Table

CREATE TABLE MESSAGES (
    Result VARCHAR2(1000)
);


2. PL/SQL Program

SET SERVEROUTPUT ON;
ACCEPT p_sal PROMPT 'Please enter the salary: ';

DECLARE
    v_sal    emp.sal%TYPE := &p_sal;
    v_ename  emp.ename%TYPE;
    v_msg    VARCHAR2(1000);
BEGIN
    -- Attempt to fetch the employee name based on the salary
    SELECT ename 
    INTO v_ename 
    FROM emp 
    WHERE sal = v_sal;

    -- If successful (exactly one row found)
    v_msg := v_ename || ' ' || v_sal;
    INSERT INTO MESSAGES (Result) VALUES (v_msg);
    DBMS_OUTPUT.PUT_LINE('Success: ' || v_msg);

EXCEPTION
    WHEN TOO_MANY_ROWS THEN
        v_msg := 'More than one employee with a salary ' || v_sal;
        INSERT INTO MESSAGES (Result) VALUES (v_msg);
        DBMS_OUTPUT.PUT_LINE(v_msg);

    WHEN NO_DATA_FOUND THEN
        v_msg := 'No Employee with that salary ' || v_sal;
        INSERT INTO MESSAGES (Result) VALUES (v_msg);
        DBMS_OUTPUT.PUT_LINE(v_msg);

    WHEN OTHERS THEN
        v_msg := 'Other Error';
        INSERT INTO MESSAGES (Result) VALUES (v_msg);
        DBMS_OUTPUT.PUT_LINE(v_msg);
END;
/


Q2) Write a program that will prompt for Empno, ename, sal and insert them into the Emp table.
Use Unnamed system defined exception handler

SOLUTION) 

DECLARE
    -- Variables to hold user input
    v_empno emp.empno%TYPE := &Employee_Number;
    v_ename emp.ename%TYPE := '&Employee_Name';
    v_sal   emp.sal%TYPE   := &Salary;
BEGIN
    -- Attempting to insert data into the Emp table
    INSERT INTO emp (empno, ename, sal)
    VALUES (v_empno, v_ename, v_sal);

    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Record inserted successfully.');

EXCEPTION
    -- Unnamed system defined exception handler
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An error occurred: ' || SQLERRM);
        ROLLBACK;
END;
/



Q3) Write a program that will prompts for a EMPNO and delete the employee from the EMP table.
Handle with User Defined Exceptions.

SOLUTIONS) 

DECLARE
    -- Declare the user-defined exception
    e_invalid_empno EXCEPTION;
    
    -- Variable to store the input employee number
    v_empno emp.empno%TYPE := &Enter_Employee_Number;
    v_count NUMBER;
BEGIN
    -- Check if the employee exists before attempting deletion
    SELECT count(*) INTO v_count FROM emp WHERE empno = v_empno;

    IF v_count = 0 THEN
        -- Raise the custom exception if ID is not found
        RAISE e_invalid_empno;
    ELSE
        -- Delete the record
        DELETE FROM emp WHERE empno = v_empno;
        DBMS_OUTPUT.PUT_LINE('Employee ' || v_empno || ' deleted successfully.');
        COMMIT;
    END IF;

EXCEPTION
    -- Handle the user-defined exception
    WHEN e_invalid_empno THEN
        DBMS_OUTPUT.PUT_LINE('Error: Employee Number ' || v_empno || ' does not exist.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An unexpected error occurred.');
        ROLLBACK;
END;
/
