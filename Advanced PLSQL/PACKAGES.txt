Q1) 1. Procedure ADD_DEPT: This procedure will take 3 IN arguments DNO, NAME, LO and duly insert these values into the DEPT table. Handle the exception if the user tries to insert a duplicate row using User Defined Exception Handler.
2. Procedure UPDATE_DEPT which takes 3 arguments DNO, NAME, LO. Update the DNAME, LOC for the corresponding DEPTNO. Handle exception when the user enters DEPTNO that does not exists.
3. Procedure DELETE_DEPT which takes 1 argument DNO. Delete the record for the corresponding DEPTNO. Handle the required exception when the user enters a not existing DEPTNO.
4. Create a Local Function CAL_CNT that returnsTRUE if DEPTNO does not Exists ELSE return FALSE.
Use this Function in all 3 Procedures. Create a One Time Procedure that displays the total number of rows.

SOLUTION) 

1. Local Function: CAL_CNTThis function checks if a department number exists. It is used as a helper in the procedures below.

CREATE OR REPLACE FUNCTION CAL_CNT(P_DNO IN DEPT.DEPTNO%TYPE) 
RETURN BOOLEAN IS
    V_COUNT NUMBER;
BEGIN
    SELECT COUNT(*) INTO V_COUNT FROM DEPT WHERE DEPTNO = P_DNO;
    IF V_COUNT = 0 THEN
        RETURN TRUE; -- Returns TRUE if DEPTNO does NOT exist
    ELSE
        RETURN FALSE;
    END IF;
END;
/


2. Procedure: ADD_DEPTInserts a new department and handles duplicates using a user-defined exception.

CREATE OR REPLACE PROCEDURE ADD_DEPT(DNO IN NUMBER, NAME IN VARCHAR2, LO IN VARCHAR2) IS
    DUP_DEPT_EXCEP EXCEPTION;
BEGIN
    -- Using the CAL_CNT function
    IF NOT CAL_CNT(DNO) THEN
        RAISE DUP_DEPT_EXCEP;
    ELSE
        INSERT INTO DEPT (DEPTNO, DNAME, LOC) VALUES (DNO, NAME, LO);
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Department added successfully.');
    END IF;
EXCEPTION
    WHEN DUP_DEPT_EXCEP THEN
        DBMS_OUTPUT.PUT_LINE('Error: Duplicate Department Number detected.');
END;
/


3. Procedure: UPDATE_DEPTUpdates an existing department; handles cases where the ID does not exist.

CREATE OR REPLACE PROCEDURE UPDATE_DEPT(DNO IN NUMBER, NAME IN VARCHAR2, LO IN VARCHAR2) IS
    NOT_FOUND_EXCEP EXCEPTION;
BEGIN
    IF CAL_CNT(DNO) THEN
        RAISE NOT_FOUND_EXCEP;
    ELSE
        UPDATE DEPT SET DNAME = NAME, LOC = LO WHERE DEPTNO = DNO;
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Department updated successfully.');
    END IF;
EXCEPTION
    WHEN NOT_FOUND_EXCEP THEN
        DBMS_OUTPUT.PUT_LINE('Error: Department Number does not exist.');
END;
/


4. Procedure: DELETE_DEPTDeletes a department record.

CREATE OR REPLACE PROCEDURE DELETE_DEPT(DNO IN NUMBER) IS
    NOT_FOUND_EXCEP EXCEPTION;
BEGIN
    IF CAL_CNT(DNO) THEN
        RAISE NOT_FOUND_EXCEP;
    ELSE
        DELETE FROM DEPT WHERE DEPTNO = DNO;
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Department deleted successfully.');
    END IF;
EXCEPTION
    WHEN NOT_FOUND_EXCEP THEN
        DBMS_OUTPUT.PUT_LINE('Error: Cannot delete. Department Number does not exist.');
END;
/

Q2) Create a Package PK2 that has 2 Procedures DISP_FIRST3 and DISP_FIRST6 that will display the first 3 employees and first 6 employees from the EMP Table. Create a Global Cursor.

SOLUTION) 

-- Package Specification
CREATE OR REPLACE PACKAGE PK2 AS
    CURSOR C_EMP IS
        SELECT * FROM EMP ORDER BY EMPNO;

    PROCEDURE DISP_FIRST3;
    PROCEDURE DISP_FIRST6;
END PK2;
/

-- Package Body
CREATE OR REPLACE PACKAGE BODY PK2 AS

    PROCEDURE DISP_FIRST3 IS
        V_EMP EMP%ROWTYPE;
    BEGIN
        OPEN C_EMP;
        FOR I IN 1..3 LOOP
            FETCH C_EMP INTO V_EMP;
            EXIT WHEN C_EMP%NOTFOUND;
            DBMS_OUTPUT.PUT_LINE(V_EMP.EMPNO || ' ' || V_EMP.ENAME);
        END LOOP;
        CLOSE C_EMP;
    END DISP_FIRST3;

    PROCEDURE DISP_FIRST6 IS
        V_EMP EMP%ROWTYPE;
    BEGIN
        OPEN C_EMP;
        FOR I IN 1..6 LOOP
            FETCH C_EMP INTO V_EMP;
            EXIT WHEN C_EMP%NOTFOUND;
            DBMS_OUTPUT.PUT_LINE(V_EMP.EMPNO || ' ' || V_EMP.ENAME);
        END LOOP;
        CLOSE C_EMP;
    END DISP_FIRST6;

END PK2;
/

Q3) Create a sequence S2 start with 1 and increment by 1. Create a package PK3 that has 2 procedures with the same name ADD_EMP. The procedure will take 3 arguments EMPNO, ENAME DEFAULT 'UNK', SAL DEFUALT 1000. This procedure inserts these values in the EMP table. The second procedure accepts 2 arguments NAME DEFAULT 'UNK', SAL DEFAULT 1000 and inserts into the EMP table calling the sequence S2 for EMPNO column. Also insert other ENAME and SAL columns.

SOLUTION) 

-- Create Sequence
CREATE SEQUENCE S2
START WITH 1
INCREMENT BY 1;
/

-- Package Specification
CREATE OR REPLACE PACKAGE PK3 AS
    PROCEDURE ADD_EMP(
        P_EMPNO NUMBER,
        P_ENAME VARCHAR2 DEFAULT 'UNK',
        P_SAL   NUMBER DEFAULT 1000
    );

    PROCEDURE ADD_EMP(
        P_ENAME VARCHAR2 DEFAULT 'UNK',
        P_SAL   NUMBER DEFAULT 1000
    );
END PK3;
/

-- Package Body
CREATE OR REPLACE PACKAGE BODY PK3 AS

    -- Procedure 1
    PROCEDURE ADD_EMP(
        P_EMPNO NUMBER,
        P_ENAME VARCHAR2 DEFAULT 'UNK',
        P_SAL   NUMBER DEFAULT 1000
    )
    IS
    BEGIN
        INSERT INTO EMP (EMPNO, ENAME, SAL)
        VALUES (P_EMPNO, P_ENAME, P_SAL);
    END;

    -- Procedure 2
    PROCEDURE ADD_EMP(
        P_ENAME VARCHAR2 DEFAULT 'UNK',
        P_SAL   NUMBER DEFAULT 1000
    )
    IS
    BEGIN
        INSERT INTO EMP (EMPNO, ENAME, SAL)
        VALUES (S2.NEXTVAL, P_ENAME, P_SAL);
    END;

END PK3;
