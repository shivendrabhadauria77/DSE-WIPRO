STORED PROCEDURES AND FUNCTIONS

Q1) Create a procedure to accept an empno as a input parameter and display the detail like ename, sal and deptno.

SOLUTION) 

CREATE OR REPLACE PROCEDURE get_employee_details (p_empno IN NUMBER)
IS
    v_ename   emp.ename%TYPE;
    v_sal     emp.sal%TYPE;
    v_deptno  emp.deptno%TYPE;
BEGIN
    -- Select the details into local variables
    SELECT ename, sal, deptno 
    INTO v_ename, v_sal, v_deptno
    FROM emp
    WHERE empno = p_empno;

    -- Display the details
    DBMS_OUTPUT.PUT_LINE('Employee Name: ' || v_ename);
    DBMS_OUTPUT.PUT_LINE('Salary: ' || v_sal);
    DBMS_OUTPUT.PUT_LINE('Department Number: ' || v_deptno);

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        DBMS_OUTPUT.PUT_LINE('Error: No employee found with ID ' || p_empno);
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An unexpected error occurred.');
END;
/

Q3) Create a procedure that will display all the names list of procedures and functions from the current schema by querying user_source table with the help of cursor.

SOLUTION) 

CREATE OR REPLACE PROCEDURE list_schema_objects AS
    -- Declare the cursor to fetch distinct names of procedures and functions
    CURSOR obj_cursor IS
        SELECT DISTINCT name, type
        FROM user_source
        WHERE type IN ('PROCEDURE', 'FUNCTION')
        ORDER BY type, name;

    -- Variables to hold the fetched data
    v_name user_source.name%TYPE;
    v_type user_source.type%TYPE;
BEGIN
    DBMS_OUTPUT.PUT_LINE('Listing Procedures and Functions:');
    DBMS_OUTPUT.PUT_LINE('---------------------------------');

    -- Open the cursor
    OPEN obj_cursor;
    LOOP
        -- Fetch data into variables
        FETCH obj_cursor INTO v_name, v_type;
        EXIT WHEN obj_cursor%NOTFOUND;

        -- Display the result
        DBMS_OUTPUT.PUT_LINE(v_type || ': ' || v_name);
    END LOOP;

    -- Close the cursor
    CLOSE obj_cursor;
END;
/
